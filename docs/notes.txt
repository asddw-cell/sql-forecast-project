Create tree filters:
  
  Forecast Type
    SELECT 
	    [Code] AS Code,
	    [Description] AS Text
    FROM [Forecast].[dbo].[tblForecastType]
  
  Sales Channel
    SELECT 
    	[Code] AS Code,
    	[Description] AS Text
    FROM [Forecast].[dbo].[tblSalesChannel]
    WHERE [Code] <> 'ECOMM'

  Year
    SELECT DISTINCT
    	[Year] AS Code,
    	[Year] AS Text
    FROM [Forecast].[dbo].[tblForecastData_US]
    WHERE [Year] >= YEAR(GETDATE())
    ORDER BY [Year] ASC

  Customer
    SELECT [Customer].[Code] AS [Code]
		,[Description] AS [Text]
	FROM [Forecast].[dbo].[tblForecastCustomer_US] [Customer]
	INNER JOIN [Forecast].[dbo].[Userlogin] [Access] ON [Customer].[Code] = [Access].[ForecastCustomer]
		AND [Access].[UserLogin] = @WindowsUser
	WHERE [Customer].[Active] = 1
	ORDER BY [Customer].[Description]
